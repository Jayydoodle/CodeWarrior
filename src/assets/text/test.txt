if(this.warriorTurn && (this.turn % 5 == 0 || this.turn == 1))
{
  this.cast("sleep", "Enemy1");
}
else if(this.warriorTurn)
  this.attack("random");
if(this.mageTurn && this.limitBurstReady)
{
  this.limitBurst("freeze");
}
else if(this.mageTurn)
  this.attack("random");
if(this.rangerTurn && this.limitBurstReady)
{
  this.limitBurst("torrent");
}
else if(this.rangerTurn)
  this.attack("random");
