if(this.warriorTurn)
{
	if(this.limitBurstReady)
  		this.limitBurst("inferno");
	else
		this.attack("random");
}
if(this.mageTurn)
{
	if(this.turn == 2 || this.turn > 1 && this.turn % 5 == 1 && this.warriorIsAlive)
		this.cast("remedy", "warrior");
	else if(this.limitBurstReady && (!this.warriorIsAlive || !this.rangerIsAlive))
		this.limitBurst("resurrection");
	else if(this.limitBurstReady)
		this.limitBurst("benediction");
	else if(this.turn % 3 == 0)
		this.cast("water", "random");
	else if(this.turn % 2 == 0)
		this.cast("fire", "random");
	else
	  	this.attack("random");
}
if(this.rangerTurn)
{
	if(this.limitBurstReady)
		this.limitBurst("tempest");
	else
  		this.attack("random");
}
